## Exemplo de Implementação

<CodeGroup>
  <Code title="JavaScript" active>
    ```javascript
    // Exemplo de geração de QR Code PIX
    async function gerarQrCode() {
      try {
        const response = await fetch('https://sandbox.ws.relaxpay.site/api/v1/gateway/request-qrcode', {
          method: 'POST',
          headers: {
            'ci': 'seu-client-id',
            'cs': 'seu-client-secret',
            'Content-Type': 'application/json'
          },
          body: JSON.stringify({
            amount: 100.00,
            dueDate: '2024-12-31',
            callbackUrl: 'https://seu-site.com/webhook/relaxpay'
          })
        });
        
        const data = await response.json();
        console.log('QR Code gerado:', data);
        
        // Exibe o QR Code para o cliente
        exibirQrCode(data.paymentCodeBase64);
      } catch (error) {
        console.error('Erro ao gerar QR Code:', error);
      }
    }
    ```
  </Code>
  <Code title="PHP">
    ```php
    <?php
    // Exemplo de geração de QR Code PIX
    function gerarQrCode() {
      $ch = curl_init();
      
      curl_setopt_array($ch, [
        CURLOPT_URL => 'https://sandbox.ws.relaxpay.site/api/v1/gateway/request-qrcode',
        CURLOPT_RETURNTRANSFER => true,
        CURLOPT_POST => true,
        CURLOPT_HTTPHEADER => [
          'ci: seu-client-id',
          'cs: seu-client-secret',
          'Content-Type: application/json'
        ],
        CURLOPT_POSTFIELDS => json_encode([
          'amount' => 100.00,
          'dueDate' => '2024-12-31',
          'callbackUrl' => 'https://seu-site.com/webhook/relaxpay'
        ])
      ]);
      
      $response = curl_exec($ch);
      $data = json_decode($response, true);
      
      // Exibe o QR Code para o cliente
      echo '<img src="data:image/png;base64,' . $data['paymentCodeBase64'] . '" />';
    }
    ```
  </Code>
  <Code title="Python">
    ```python
    import requests
    import json
    
    # Exemplo de geração de QR Code PIX
    def gerar_qr_code():
        try:
            response = requests.post(
                'https://sandbox.ws.relaxpay.site/api/v1/gateway/request-qrcode',
                headers={
                    'ci': 'seu-client-id',
                    'cs': 'seu-client-secret',
                    'Content-Type': 'application/json'
                },
                json={
                    'amount': 100.00,
                    'dueDate': '2024-12-31',
                    'callbackUrl': 'https://seu-site.com/webhook/relaxpay'
                }
            )
            
            data = response.json()
            print('QR Code gerado:', data['idTransaction'])
            
            # Exibe o QR Code para o cliente
            return data['paymentCodeBase64']
        except Exception as e:
            print('Erro ao gerar QR Code:', str(e))
    ```
  </Code>
  <Code title="C#">
    ```csharp
    using System;
    using System.Net.Http;
    using System.Text;
    using System.Threading.Tasks;
    using Newtonsoft.Json;
    
    public class RelaxPayClient
    {
        private readonly HttpClient _client = new HttpClient();
        
        public async Task<string> GerarQrCodeAsync()
        {
            var request = new
            {
                amount = 100.00,
                dueDate = "2024-12-31",
                callbackUrl = "https://seu-site.com/webhook/relaxpay"
            };
            
            var content = new StringContent(
                JsonConvert.SerializeObject(request),
                Encoding.UTF8,
                "application/json");
                
            _client.DefaultRequestHeaders.Add("ci", "seu-client-id");
            _client.DefaultRequestHeaders.Add("cs", "seu-client-secret");
            
            var response = await _client.PostAsync(
                "https://sandbox.ws.relaxpay.site/api/v1/gateway/request-qrcode", 
                content);
                
            var responseContent = await response.Content.ReadAsStringAsync();
            dynamic data = JsonConvert.DeserializeObject(responseContent);
            
            // Exibe o QR Code para o cliente
            return data.paymentCodeBase64;
        }
    }
    ```

    Realize testes completos no ambiente sandbox antes de migrar para produção
    
    <CodeGroup>
      <Code title="Teste de Webhook" active>
        ```bash
        # Substitua com seu Client Secret
        SECRET="seu-client-secret"
        
        # Crie o payload
        PAYLOAD='{"event":"payment.received","data":{"idTransaction":"test-123"}}'
        
        # Calcule a assinatura
        SIGNATURE=$(echo -n "$PAYLOAD" | openssl dgst -sha256 -hmac "$SECRET" | awk '{print $2}')
        
        # Envie para seu endpoint
        curl -X POST https://seu-site.com/webhook/relaxpay \
          -H "Content-Type: application/json" \
          -H "X-Signature: $SIGNATURE" \
          -d "$PAYLOAD"
        ```
      </Code>
    </CodeGroup>
  </Step>
  <Step title="Produção" icon="rocket">
    Após validar sua integração, solicite credenciais de produção e atualize os endpoints em sua aplicação
  </Step>
</Steps>
